import { StarCount } from "@components"

export const getStaticProps = async () => {
  const propsModel = { props: { ssg: { starCount: 0 } }, revalidate: 60 }
  try {
    const res = await fetch("https://api.github.com/repos/recallwei/recallwei")
    const data = await res.json()
    propsModel.props.ssg.starCount = data.stargazers_count
    return propsModel
  } catch {
    return propsModel
  }
}
