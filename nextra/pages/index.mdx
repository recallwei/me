import { StarCount } from "../components"

export const getStaticProps = async () => {
  const res = await fetch("https://api.github.com/repos/recallwei/recallwei")
  const data = await res.json()
  return {
    props: {
      ssg: {
        starCount: data.stargazers_count
      }
    },
    revalidate: 60
  }
}
