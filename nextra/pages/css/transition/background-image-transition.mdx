# Background Image Transition

Chrome does support background image transition. But currently it is not working on the other Browser like FireFox ( 2022-12-31 ).

```css showLineNumbers copy
.bg-image {
  background-image: url(/images/wallpaper1.jpg);
  backgroundrepeat: no-repeat;
  backgroundsize: cover;
  transition: background-image 2s ease;
}
```

import React from "react"
import wallpaper1 from "public/images/wallpaper1.jpg"
import wallpaper2 from "public/images/wallpaper2.jpg"

export const BgImageShowCase = () => {
  const [styleFlag, setStyleFlag] = React.useState(true)
  const loading = React.useRef(false)
  const getStyle = (type) => ({
    backgroundImage: `url(/images/wallpaper${type}.jpg)`,
    backgroundRepeat: "no-repeat",
    backgroundSize: "cover",
    transition: "background-image 2s ease",
    cursor: "pointer",
    width: "100%",
    height: "400px",
    borderRadius: "0.75rem"
  })
  const onClickImage = () => {
    if (loading.current) return
    loading.current = true
    setTimeout(() => {
      loading.current = false
    }, 2000)
    setStyleFlag(!styleFlag)
  }
  return (
    <div
      style={getStyle(styleFlag ? 1 : 2)}
      onClick={onClickImage}
    />
  )
}

<BgImageShowCase />
