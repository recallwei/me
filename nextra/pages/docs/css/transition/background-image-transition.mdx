# Background Image Transition

## Compatibility

**Chrome** does support CSS background image transition. But currently it is not working on the other Browser like **FireFox** ( 2022-12-31 ).

```css filename=".css" showLineNumbers  {2,5}
.bg-image {
  background-image: url(/images/wallpaper1.jpg);
  backgroundrepeat: no-repeat;
  backgroundsize: cover;
  transition: background-image 2s ease;
}
```

import { useState } from "react"
export const BgImageShowCase = () => {
  const [styleFlag, setStyleFlag] = useState(true)
  const [loading, setLoading] = useState(false)
  const getStyle = (type) => ({
    backgroundImage: `url(/images/wallpaper${type}.jpg)`,
    transition: "background-image 2s ease"
  })
  const onClickImage = () => {
    if (loading) return
    setLoading(true)
    setTimeout(() => setLoading(false), 2000)
    setStyleFlag(!styleFlag)
  }
  return (
    <div
      style={{ ...getStyle(styleFlag ? 1 : 2), cursor: loading ? "wait" : "pointer" }}
      class="rounded-xl h-96 mt-4 w-full bg-cover bg-no-repeat"
      onClick={onClickImage}
    />
  )
}
